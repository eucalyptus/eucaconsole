delete-volume-from-lp
setSpeed 500
assertVisible id=resource-menu-dashboard
setTimeout 150000
clickAndWait id=resource-menu-dashboard
waitForVisible css=#volumes > div.tile > div.content > a > i.icon
clickAndWait css=#volumes > div.tile > div.content > a > i.icon
waitForVisible //span/a[@id="table-item-dropdown_${volume_id}"]/i
click //span/a[@id="table-item-dropdown_${volume_id}"]/i
waitForVisible //ul[@id="item-dropdown_${volume_id}"]/li[5]
clickAt //ul[@id="item-dropdown_${volume_id}"]/li[5]/a
assertText //div[@id='delete-volume-modal']/p[2]/span[2]/b ${volume_id}
click id=delete_volume_submit_button
storeEval 1 k
waitForVisible //td/a[@href="/volumes/${volume_id}"]/../../td[2]/span
storeText //td/a[@href="/volumes/${volume_id}"]/../../td[2]/span status
echo javascript{storedVars['status']=${status}};
echo ${status}
storeEval storedVars['deleted']="deleted";
echo ${deleted}
storeEval storedVars['status']==storedVars['deleted'] is_deleted
echo ${is_deleted}
echo javascript{storedVars['is_deleted']=${is_deleted}};
while ${k}<20&&!${is_deleted}
pause 30000
storeText //td/a[@href="/volumes/${volume_id}"]/../../td[2]/span status
echo ${status}
storeEval storedVars['status']==storedVars['deleted'] is_deleted
echo ${is_deleted}
storeEval ${k}+1 k
endWhile
assertText //td/a[@href="/volumes/${volume_id}"]/../../td[2]/span deleted
